24 /// A namespace that is used to store existing names and generate new names in
25 /// some scope within the IR. This exists to work around limitations of
26 /// SymbolTables. This acts as a base class providing facilities common to all
27 /// namespaces implementations.
28 class Namespace {
29 public:
30  Namespace() {}
31  Namespace(const Namespace &other) = default;
32  Namespace(Namespace &&other) : nextIndex(std::move(other.nextIndex)) {}
33 
34  Namespace &operator=(const Namespace &other) = default;
35  Namespace &operator=(Namespace &&other) {
36  nextIndex = std::move(other.nextIndex);
37  return *this;
38  }
39 
40  /// Empty the namespace.
41  void clear() { nextIndex.clear(); }
42 
43  /// Return a unique name, derived from the input `name`, and add the new name
44  /// to the internal namespace. There are two possible outcomes for the
45  /// returned name:
46  ///
47  /// 1. The original name is returned.
48  /// 2. The name is given a `_<n>` suffix where `<n>` is a number starting from
49  /// `_0` and incrementing by one each time.
50  StringRef newName(const Twine &name) {
51  // Special case the situation where there is no name collision to avoid
52  // messing with the SmallString allocation below.
53  llvm::SmallString<64> tryName;
54  auto inserted = nextIndex.insert({name.toStringRef(tryName), 0});
55  if (inserted.second)
56  return inserted.first->getKey();
57 
58  // Try different suffixes until we get a collision-free one.
59  if (tryName.empty())
60  name.toVector(tryName); // toStringRef may leave tryName unfilled
61 
62  // Indexes less than nextIndex[tryName] are lready used, so skip them.
63  // Indexes larger than nextIndex[tryName] may be used in another name.
64  size_t &i = nextIndex[tryName];
65  tryName.push_back('_');
66  size_t baseLength = tryName.size();
67  do {
68  tryName.resize(baseLength);
69  Twine(i++).toVector(tryName); // append integer to tryName
70  inserted = nextIndex.insert({tryName, 0});
71  } while (!inserted.second);
72 
73  return inserted.first->getKey();
74  }
75 
76 protected:
77  // The "next index" that will be tried when trying to unique a string within a
78  // namespace. It follows that all values less than the "next index" value are
79  // already used.
80  llvm::StringMap<size_t> nextIndex;
81 };
