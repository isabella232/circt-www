29 FieldRef circt::firrtl::getFieldRefFromValue(Value value) {
30  // This code walks upwards from the subfield and calculates the field ID at
31  // each level. At each stage, it must take the current id, and re-index it as
32  // a nested bundle under the parent field.. This is accomplished by using the
33  // parent field's ID as a base, and adding the field ID of the child.
34  unsigned id = 0;
35  while (value) {
36  Operation *op = value.getDefiningOp();
37 
38  // If this is a block argument, we are done.
39  if (!op)
40  break;
41 
42  if (auto subfieldOp = dyn_cast<SubfieldOp>(op)) {
43  value = subfieldOp.input();
44  auto bundleType = value.getType().cast<BundleType>();
45  // Rebase the current index on the parent field's index.
46  id += bundleType.getFieldID(subfieldOp.fieldIndex());
47  } else if (auto subindexOp = dyn_cast<SubindexOp>(op)) {
48  value = subindexOp.input();
49  auto vecType = value.getType().cast<FVectorType>();
50  // Rebase the current index on the parent field's index.
51  id += vecType.getFieldID(subindexOp.index());
52  } else {
53  break;
54  }
55  }
56  return {value, id};
57 }

59 /// Get the string name of a value which is a direct child of a declaration op.
60 static void getDeclName(Value value, SmallString<64> &string) {
61  if (auto arg = value.dyn_cast<BlockArgument>()) {
62  // Get the module ports and get the name.
63  auto module = cast<FModuleOp>(arg.getOwner()->getParentOp());
64  SmallVector<PortInfo> ports = module.getPorts();
65  string += ports[arg.getArgNumber()].name.getValue();
66  return;
67  }
68 
69  auto *op = value.getDefiningOp();
70  TypeSwitch<Operation *>(op)
71  .Case<InstanceOp, MemOp>([&](auto op) {
72  string += op.name();
73  string += ".";
74  string +=
75  op.getPortName(value.cast<OpResult>().getResultNumber()).getValue();
76  })
77  .Case<WireOp, RegOp, RegResetOp>([&](auto op) { string += op.name(); });
78 }

85 std::string circt::firrtl::getFieldName(const FieldRef &fieldRef,
86  bool &rootKnown) {
87  SmallString<64> name;
88  auto value = fieldRef.getValue();
89  getDeclName(value, name);
90  rootKnown = !name.empty();
91 
92  auto type = value.getType();
93  auto localID = fieldRef.getFieldID();
94  while (localID) {
95  if (auto bundleType = type.dyn_cast<BundleType>()) {
96  auto index = bundleType.getIndexForFieldID(localID);
97  // Add the current field string, and recurse into a subfield.
98  auto &element = bundleType.getElements()[index];
99  if (!name.empty())
100  name += ".";
101  name += element.name.getValue();
102  // Recurse in to the element type.
103  type = element.type;
104  localID = localID - bundleType.getFieldID(index);
105  } else if (auto vecType = type.dyn_cast<FVectorType>()) {
106  auto index = vecType.getIndexForFieldID(localID);
107  name += "[";
108  name += std::to_string(index);
109  name += "]";
110  // Recurse in to the element type.
111  type = vecType.getElementType();
112  localID = localID - vecType.getFieldID(index);
113  } else {
114  // If we reach here, the field ref is pointing inside some aggregate type
115  // that isn't a bundle or a vector. If the type is a ground type, then the
116  // localID should be 0 at this point, and we should have broken from the
117  // loop.
118  llvm_unreachable("unsupported type");
119  }
120  }
121 
122  return name.str().str();
123 }

141 /// Registered hook to materialize a single constant operation from a given
142 /// attribute value with the desired resultant type. This method should use
143 /// the provided builder to create the operation without changing the
144 /// insertion position. The generated operation is expected to be constant
145 /// like, i.e. single result, zero operands, non side-effecting, etc. On
146 /// success, this hook should return the value generated to represent the
147 /// constant value. Otherwise, it should return null on failure.
148 Operation *FIRRTLDialect::materializeConstant(OpBuilder &builder,
149  Attribute value, Type type,
150  Location loc) {
151 
152  // Boolean constants. Boolean attributes are always a special constant type
153  // like ClockType and ResetType. Since BoolAttrs are also IntegerAttrs, its
154  // important that this goes first.
155  if (auto attrValue = value.dyn_cast<BoolAttr>()) {
156  assert((type.isa<ClockType>() || type.isa<AsyncResetType>() ||
157  type.isa<ResetType>()) &&
158  "BoolAttrs can only be materialized for special constant types.");
159  return builder.create<SpecialConstantOp>(loc, type, attrValue);
160  }
161 
162  // Integer constants.
163  if (auto attrValue = value.dyn_cast<IntegerAttr>()) {
164  // Integer attributes (ui1) might still be special constant types.
165  if (attrValue.getValue().getBitWidth() == 1 &&
166  (type.isa<ClockType>() || type.isa<AsyncResetType>() ||
167  type.isa<ResetType>()))
168  return builder.create<SpecialConstantOp>(
169  loc, type,
170  builder.getBoolAttr(attrValue.getValue().isAllOnesValue()));
171 
172  assert((!type.cast<IntType>().hasWidth() ||
173  (unsigned)type.cast<IntType>().getWidthOrSentinel() ==
174  attrValue.getValue().getBitWidth()) &&
175  "type/value width mismatch materializing constant");
176  return builder.create<ConstantOp>(loc, type, attrValue);
177  }
178 
179  // InvalidValue constants.
180  if (auto invalidValue = value.dyn_cast<InvalidValueAttr>())
181  return builder.create<InvalidValueOp>(loc, type);
182 
183  return nullptr;
184 }
