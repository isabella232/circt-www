21 /// This stores lookup tables to make manipulating and working with the IR more
22 /// efficient. There are two phases to this object: the "building" phase in
23 /// which it is "write only" and then the "using" phase which is read-only (and
24 /// thus can be used by multiple threads). The
25 /// "freeze" method transitions between the two states.
26 class SymbolCache {
27 public:
28  class Item {
29  public:
30  Item(mlir::Operation *op) : op(op), port(~0ULL) {}
31  Item(mlir::Operation *op, size_t port) : op(op), port(port) {}
32  bool hasPort() const { return port != ~0ULL; }
33  mlir::Operation *getOp() const { return op; }
34  size_t getPort() const { return port; }
35 
36  private:
37  mlir::Operation *op;
38  size_t port;
39  };

41  /// In the building phase, add symbols.
42  void addDefinition(mlir::StringAttr symbol, mlir::Operation *op) {
43  assert(!isFrozen && "cannot mutate a frozen cache");
44  symbolCache.try_emplace(symbol, op, ~0ULL);
45  }
46 
47  // Add inner names, which might be ports
48  void addDefinition(mlir::StringAttr modSymbol, mlir::StringAttr name,
49  mlir::Operation *op, size_t port = ~0ULL) {
50  assert(!isFrozen && "cannot mutate a frozen cache");
51  auto key = InnerRefAttr::get(modSymbol.getContext(), modSymbol, name);
52  symbolCache.try_emplace(key, op, port);
53  }

55  // Add inner names, which might be ports
56  void addDefinition(InnerRefAttr name, mlir::Operation *op,
57  size_t port = ~0ULL) {
58  assert(!isFrozen && "cannot mutate a frozen cache");
59  symbolCache.try_emplace(name, op, port);
60  }

62  /// Mark the cache as frozen, which allows it to be shared across threads.
63  void freeze() { isFrozen = true; }

77  Item getDefinition(InnerRefAttr name) const { return lookup(name); }

79 private:
80  Item lookup(InnerRefAttr attr) const {
81  assert(isFrozen && "cannot read from this cache until it is frozen");
82  auto it = symbolCache.find(attr);
83  return it == symbolCache.end() ? Item{nullptr, ~0ULL} : it->second;
84  }
85 
86  mlir::Operation *lookup(mlir::StringAttr attr) const {
87  assert(isFrozen && "cannot read from this cache until it is frozen");
88  auto it = symbolCache.find(attr);
89  if (it == symbolCache.end())
90  return nullptr;
91  assert(!it->second.hasPort() && "Module names should never be ports");
92  return it->second.getOp();
93  }
94 
95  bool isFrozen = false;
96 
97  /// This stores a lookup table from symbol attribute to the operation
98  /// (hw.module, hw.instance, etc) that defines it.
99  /// TODO: It is super annoying that symbols are *defined* as StringAttr, but
100  /// are then referenced as FlatSymbolRefAttr. Why can't we have nice
101  /// pointer uniqued things?? :-(
102  llvm::DenseMap<mlir::Attribute, Item> symbolCache;
103 };
