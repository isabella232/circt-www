102 /// Walk all the placements within 'bounds' ([xmin, xmax, ymin, ymax], inclusive
103 /// on all sides), with -1 meaning unbounded.
104 MLIR_CAPI_EXPORTED void circtMSFTPlacementDBWalkPlacements(
105  CirctMSFTPlacementDB cdb, CirctMSFTPlacementCallback ccb, int64_t bounds[4],
106  CirctMSFTPrimitiveType cPrimTypeFilter, CirctMSFTWalkOrder cWalkOrder,
107  void *userData) {
108 
109  PlacementDB *db = unwrap(cdb);
110  auto cb = [ccb, userData](PhysLocationAttr loc, PDPhysLocationOp locOp) {
111  ccb(wrap(loc), wrap(locOp), userData);
112  };
113  Optional<PrimitiveType> primTypeFilter;
114  if (cPrimTypeFilter >= 0)
115  primTypeFilter = static_cast<PrimitiveType>(cPrimTypeFilter);
116 
117  Optional<PlacementDB::WalkOrder> walkOrder;
118  if (cWalkOrder.columns != CirctMSFTDirection::NONE ||
119  cWalkOrder.rows != CirctMSFTDirection::NONE)
120  walkOrder = PlacementDB::WalkOrder{
121  static_cast<PlacementDB::Direction>(cWalkOrder.columns),
122  static_cast<PlacementDB::Direction>(cWalkOrder.rows)};
123 
124  db->walkPlacements(
125  cb, std::make_tuple(bounds[0], bounds[1], bounds[2], bounds[3]),
126  primTypeFilter, walkOrder);
127 }
