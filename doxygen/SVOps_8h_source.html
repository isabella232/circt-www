39 /// This describes the bit in a pattern, 0/1/x.
40 enum class CasePatternBit { Zero = 0, One = 1, Any = 2 };

42 /// Return the letter for the specified pattern bit, e.g. "0", "1", "?" or "x".
43 /// isVerilog indicates whether we should use "?" (verilog syntax) or "x" (mlir
44 /// operation syntax.

47 // This is provides convenient access to encode and decode a pattern.
48 struct CasePattern {
49  IntegerAttr attr;
50 
51  // Return the number of bits in the pattern.
52  size_t getWidth() const { return attr.getValue().getBitWidth() / 2; }
53 
54  /// Return the specified bit, bit 0 is the least significant bit.
55  CasePatternBit getBit(size_t bitNumber) const;
56 
57  /// Return true if this pattern always matches.
58  bool isDefault() const;
59 
60  /// Get a CasePattern from a specified list of CasePatternBit. Bits are
61  /// specified in most least significant order - element zero is the least
62  /// significant bit.
63  CasePattern(ArrayRef<CasePatternBit> bits, MLIRContext *context);
64 
65  /// Get a CasePattern for the specified constant value.
66  CasePattern(const APInt &value, MLIRContext *context);
67 
68  /// Get a CasePattern with a correctly encoded attribute.
69  CasePattern(IntegerAttr attr) : attr(attr) {}
70 
71  static CasePattern getDefault(unsigned width, MLIRContext *context);
