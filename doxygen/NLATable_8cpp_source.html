16 NLATable::NLATable(Operation *operation) {
17  if (auto mod = dyn_cast<mlir::ModuleOp>(operation))
18  for (auto &op : *mod.getBody())
19  if ((operation = dyn_cast<CircuitOp>(&op)))
20  break;
21 
22  auto circuit = cast<CircuitOp>(operation);
23  // We are assuming it's faster to iterate over the top level twice than cache
24  // a large number of options.
25  for (auto &op : *circuit.getBody()) {
26  if (auto module = dyn_cast<FModuleLike>(op))
27  symToOp[module.moduleNameAttr()] = module;
28  if (auto nla = dyn_cast<NonLocalAnchor>(op)) {
29  symToOp[nla.sym_nameAttr()] = nla;
30  for (auto ent : nla.namepath()) {
31  if (auto mod = ent.dyn_cast<FlatSymbolRefAttr>())
32  nodeMap[mod.getAttr()].push_back(nla);
33  else if (auto inr = ent.dyn_cast<hw::InnerRefAttr>())
34  nodeMap[inr.getModule()].push_back(nla);
35  }
36  }
37  }
38 }

40 void NLATable::renameModule(StringAttr oldModName, StringAttr newModName) {
41  auto op = symToOp.find(oldModName);
42  if (op == symToOp.end())
43  return;
44  auto iter = nodeMap.find(oldModName);
45  if (iter == nodeMap.end())
46  return;
47  for (auto nla : iter->second)
48  nla.updateModule(oldModName, newModName);
49  nodeMap[newModName] = iter->second;
50  nodeMap.erase(oldModName);
51  symToOp[newModName] = op->second;
52  symToOp.erase(oldModName);
53 }

63 void NLATable::updateModuleInNLA(NonLocalAnchor nlaOp, StringAttr oldModule,
64  StringAttr newModule) {
65  nlaOp.updateModule(oldModule, newModule);
66  auto &nlas = nodeMap[oldModule];
67  auto *iter = std::find(nlas.begin(), nlas.end(), nlaOp);
68  if (iter != nlas.end()) {
69  nlas.erase(iter);
70  if (nlas.empty())
71  nodeMap.erase(oldModule);
72  nodeMap[newModule].push_back(nlaOp);
73  }
74 }
