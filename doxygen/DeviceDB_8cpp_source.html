84 /// Assign an instance to a primitive. Return false if another instance is
85 /// already placed at that location
86 PDPhysLocationOp PlacementDB::place(DynamicInstanceOp inst,
87  PhysLocationAttr loc, StringRef subPath,
88  Location srcLoc) {
89  StringAttr subPathAttr;
90  if (!subPath.empty())
91  subPathAttr = StringAttr::get(inst->getContext(), subPath);
92  PDPhysLocationOp locOp =
93  OpBuilder(inst.body())
94  .create<PDPhysLocationOp>(srcLoc, loc, subPathAttr,
95  FlatSymbolRefAttr());
96  if (succeeded(insertPlacement(locOp)))
97  return locOp;
98  locOp->erase();
99  return {};
100 }

102 LogicalResult PlacementDB::insertPlacement(PDPhysLocationOp locOp) {
103  PlacementCell *leaf = getLeaf(locOp.loc());
104  if (!leaf)
105  return locOp->emitOpError("Could not apply placement. Invalid location: ")
106  << locOp.loc();
107  if (leaf->locOp != nullptr)
108  return locOp->emitOpError("Could not apply placement ")
109  << locOp.loc() << ". Position already occupied by "
110  << cast<DynamicInstanceOp>(leaf->locOp->getParentOp()).appid();
111 
112  leaf->locOp = locOp;
113  return success();
114 }

131 /// Using the operation attributes, add the proper placements to the database.
132 /// Return the number of placements which weren't added due to conflicts.
133 size_t PlacementDB::addPlacements(DynamicInstanceOp inst) {
134  size_t numFailed = 0;
135  inst->walk([&](Operation *op) {
136  LogicalResult added =
137  TypeSwitch<Operation *, LogicalResult>(op)
138  .Case([&](PDPhysLocationOp op) { return insertPlacement(op); })
139  .Case([&](PDPhysRegionOp op) {
140  regionPlacements.push_back(op);
141  return success();
142  })
143  .Default([](Operation *op) { return failure(); });
144  if (failed(added))
145  ++numFailed;
146  });
147 
148  return numFailed;
149 }

169 /// Move the placement at a given location to a new location. Returns failure
170 /// if nothing was placed at the previous location or something is already
171 /// placed at the new location.
172 LogicalResult PlacementDB::movePlacement(PDPhysLocationOp locOp,
173  PhysLocationAttr newLoc) {
174  PhysLocationAttr oldLoc = locOp.locAttr();
175  PlacementCell *oldLeaf = getLeaf(oldLoc);
176  PlacementCell *newLeaf = getLeaf(newLoc);
177 
178  if (!oldLeaf || !newLeaf)
179  return failure();
180 
181  if (oldLeaf->locOp == nullptr)
182  return locOp.emitError("cannot move from a location not occupied by "
183  "specified op. Currently unoccupied");
184  if (oldLeaf->locOp != locOp)
185  return locOp.emitError("cannot move from a location not occupied by "
186  "specified op. Currently occupied by ")
187  << oldLeaf->locOp;
188  if (newLeaf->locOp)
189  return locOp.emitError(
190  "cannot move to new location since location is occupied by ")
191  << cast<DynamicInstanceOp>(newLeaf->locOp->getParentOp()).appid();
192 
193  locOp.locAttr(newLoc);
194  newLeaf->locOp = locOp;
195  oldLeaf->locOp = {};
196  return success();
197 }

210 PhysLocationAttr PlacementDB::getNearestFreeInColumn(PrimitiveType prim,
211  uint64_t columnNum,
212  uint64_t nearestToY) {
213  // Simplest possible algorithm.
214  PhysLocationAttr nearest = {};
215  walkPlacements(
216  [&nearest, nearestToY](PhysLocationAttr loc, PDPhysLocationOp locOp) {
217  if (locOp)
218  return;
219  if (!nearest) {
220  nearest = loc;
221  return;
222  }
223  int64_t curDist =
224  std::abs((int64_t)nearestToY - (int64_t)nearest.getY());
225  int64_t replDist = std::abs((int64_t)nearestToY - (int64_t)loc.getY());
226  if (replDist < curDist)
227  nearest = loc;
228  },
229  std::make_tuple(columnNum, columnNum, -1, -1), prim);
230  return nearest;
231 }

233 PlacementDB::PlacementCell *PlacementDB::getLeaf(PhysLocationAttr loc) {
234  PrimitiveType primType = loc.getPrimitiveType().getValue();
235 
236  DimNumMap &nums = placements[loc.getX()][loc.getY()];
237  if (!seeded)
238  return &nums[loc.getNum()][primType];
239  if (!nums.count(loc.getNum()))
240  return {};
241 
242  DimDevType &primitives = nums[loc.getNum()];
243  if (primitives.count(primType) == 0)
244  return {};
245  return &primitives[primType];
246 }

248 /// Walker for placements.
249 void PlacementDB::walkPlacements(
250  function_ref<void(PhysLocationAttr, PDPhysLocationOp)> callback,
251  std::tuple<int64_t, int64_t, int64_t, int64_t> bounds,
252  Optional<PrimitiveType> primType, Optional<WalkOrder> walkOrder) {
253  uint64_t xmin = std::get<0>(bounds) < 0 ? 0 : std::get<0>(bounds);
254  uint64_t xmax = std::get<1>(bounds) < 0 ? std::numeric_limits<uint64_t>::max()
255  : (uint64_t)std::get<1>(bounds);
256  uint64_t ymin = std::get<2>(bounds) < 0 ? 0 : std::get<2>(bounds);
257  uint64_t ymax = std::get<3>(bounds) < 0 ? std::numeric_limits<uint64_t>::max()
258  : (uint64_t)std::get<3>(bounds);
259 
260  // TODO: Since the data structures we're using aren't sorted, the best we can
261  // do is iterate and filter. If a specific order is requested, we sort the
262  // keys by that as we go. Once we get to performance, we'll figure out the
263  // right data structure.
264 
265  auto maybeSort = [](auto &container, auto direction) {
266  if (!direction.hasValue())
267  return;
268  if (*direction == Direction::NONE)
269  return;
270 
271  llvm::sort(container, [direction](auto colA, auto colB) {
272  if (*direction == Direction::ASC)
273  return colA.first < colB.first;
274 
275  return colA.first > colB.first;
276  });
277  };
278 
279  // X loop.
280  SmallVector<std::pair<size_t, DimYMap>> cols(placements.begin(),
281  placements.end());
282  maybeSort(cols, walkOrder.map([](auto wo) { return wo.columns; }));
283  for (auto colF : cols) {
284  size_t x = colF.first;
285  if (x < xmin || x > xmax)
286  continue;
287  DimYMap yMap = colF.second;
288 
289  // Y loop.
290  SmallVector<std::pair<size_t, DimNumMap>> rows(yMap.begin(), yMap.end());
291  maybeSort(rows, walkOrder.map([](auto wo) { return wo.rows; }));
292  for (auto rowF : rows) {
293  size_t y = rowF.first;
294  if (y < ymin || y > ymax)
295  continue;
296  DimNumMap numMap = rowF.second;
297 
298  // Num loop.
299  for (auto numF = numMap.begin(), numE = numMap.end(); numF != numE;
300  ++numF) {
301  size_t num = numF->getFirst();
302  DimDevType devMap = numF->getSecond();
303 
304  // DevType loop.
305  for (auto devF = devMap.begin(), devE = devMap.end(); devF != devE;
306  ++devF) {
307  PrimitiveType devtype = devF->getFirst();
308  if (primType && devtype != *primType)
309  continue;
310  PlacementCell &inst = devF->getSecond();
311 
312  // Marshall and run the callback.
313  PhysLocationAttr loc = PhysLocationAttr::get(
314  ctxt, PrimitiveTypeAttr::get(ctxt, devtype), x, y, num);
315  callback(loc, inst.locOp);
316  }
317  }
318  }
319  }
320 }
