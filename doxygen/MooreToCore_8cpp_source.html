150 template <typename Op>
151 void addGenericLegality(ConversionTarget &target) {
152  target.addDynamicallyLegalOp<Op>([](Op op) {
153  return !hasMooreType(op->getOperands()) && !hasMooreType(op->getResults());
154  });

157 static void populateLegality(ConversionTarget &target) {
158  target.addIllegalDialect<MooreDialect>();
159  target.addLegalDialect<mlir::BuiltinDialect>();
160  target.addLegalDialect<hw::HWDialect>();
161  target.addLegalDialect<llhd::LLHDDialect>();
162  target.addLegalDialect<comb::CombDialect>();
163 
164  addGenericLegality<cf::CondBranchOp>(target);
165  addGenericLegality<cf::BranchOp>(target);
166  addGenericLegality<func::CallOp>(target);
167  addGenericLegality<func::ReturnOp>(target);
168 
169  target.addDynamicallyLegalOp<FuncOp>([](FuncOp op) {
170  auto argsConverted = llvm::none_of(op.getBlocks(), [](auto &block) {
171  return hasMooreType(block.getArguments());
172  });
173  auto resultsConverted = !hasMooreType(op.getResultTypes());
174  return argsConverted && resultsConverted;
175  });
176 }

178 static void populateTypeConversion(TypeConverter &typeConverter) {
179  typeConverter.addConversion([&](IntType type) {
180  return mlir::IntegerType::get(type.getContext(), type.getBitSize());
181  });
182  typeConverter.addConversion([&](LValueType type) {
183  auto inner = typeConverter.convertType(type.getNestedType());
184  return llhd::SigType::get(inner);
185  });
186 
187  // Valid target types.
188  typeConverter.addConversion([](mlir::IntegerType type) { return type; });
189 }

217 /// This is the main entrypoint for the Moore to Core conversion pass.
218 void MooreToCorePass::runOnOperation() {
219  MLIRContext &context = getContext();
220  ModuleOp module = getOperation();
221 
222  ConversionTarget target(context);
223  TypeConverter typeConverter;
224  RewritePatternSet patterns(&context);
225  populateLegality(target);
226  populateTypeConversion(typeConverter);
227  populateOpConversion(patterns, typeConverter);
228 
229  if (failed(applyFullConversion(module, target, std::move(patterns))))
230  signalPassFailure();
231 }
