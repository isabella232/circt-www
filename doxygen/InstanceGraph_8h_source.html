25 /// This just maps a iterator of references to an iterator of addresses.
26 template <typename It>
27 struct AddressIterator
28  : public llvm::mapped_iterator<It, typename It::pointer (*)(
29  typename It::reference)> {
30  // This using statement is to get around a bug in MSVC. Without it, it
31  // tries to look up "It" as a member type of the parent class.
32  using Iterator = It;
33  /* implicit */ AddressIterator(Iterator iterator)
34  : llvm::mapped_iterator<It, typename Iterator::pointer (*)(
35  typename Iterator::reference)>(
36  iterator, &std::addressof<typename Iterator::value_type>) {}
37 };

42 /// This is an edge in the InstanceGraph. This tracks a specific instantiation
43 /// of a module.
44 class InstanceRecord
45  : public llvm::ilist_node_with_parent<InstanceRecord, InstanceGraphNode> {

52

55

58  void erase();

60 private:

67  InstanceRecord(const InstanceRecord &) = delete;

69  /// This is the module where the InstanceOp lives.
70  InstanceGraphNode *parent;
71 
72  /// The InstanceOp that this is tracking.
73  InstanceOp instance;
74 
75  /// This is the module which the InstanceOp is instantiating.
76  InstanceGraphNode *target;
77  /// Intrusive linked list for other uses.
78  InstanceRecord *nextUse = nullptr;

82 /// This is a Node in the InstanceGraph. Each node represents a Module in a
83 /// Circuit. Both external modules and regular modules can be represented by
84 /// this class. It is possible to efficiently iterate all modules instantiated
85 /// by this module, as well as all instantiations of this module.
86 class InstanceGraphNode : public llvm::ilist_node<InstanceGraphNode> {
87  using InstanceList = llvm::iplist<InstanceRecord>;
88 
89 public:
90  /// Get the module that this node is tracking.
91  Operation *getModule() const { return module; }
92 
93  /// Iterate the instance records in this module.
94  using iterator = detail::AddressIterator<InstanceList::iterator>;
95  iterator begin() { return instances.begin(); }
96  iterator end() { return instances.end(); }
97 
98  /// Return true if there are no more instances of this module.
99  bool noUses() { return !firstUse; }
100 
101  /// Return true if this module has exactly one use.
102  bool hasOneUse() { return llvm::hasSingleElement(uses()); }
103 
104  /// Get the number of direct instantiations of this module.
105  size_t getNumUses() { return std::distance(usesBegin(), usesEnd()); }
106 
107  /// Iterator for module uses.
108  struct UseIterator
109  : public llvm::iterator_facade_base<
110  UseIterator, std::forward_iterator_tag, InstanceRecord *> {
111  UseIterator() : current(nullptr) {}
112  UseIterator(InstanceGraphNode *node) : current(node->firstUse) {}
113  InstanceRecord *operator*() const { return current; }
114  using llvm::iterator_facade_base<UseIterator, std::forward_iterator_tag,
115  InstanceRecord *>::operator++;
116  UseIterator &operator++() {
117  assert(current && "incrementing past end");
118  current = current->nextUse;
119  return *this;
120  }
121  bool operator==(const UseIterator &other) const {
122  return current == other.current;
123  }
124 
125  private:
126  InstanceRecord *current;
127  };
128 
129  /// Iterate the instance records which instantiate this module.
130  UseIterator usesBegin() { return {this}; }
131  UseIterator usesEnd() { return {}; }
132  llvm::iterator_range<UseIterator> uses() {
133  return llvm::make_range(usesBegin(), usesEnd());
134  }
135 
136  /// Record a new instance op in the body of this module. Returns a newly
137  /// allocated InstanceRecord which will be owned by this node.
138  InstanceRecord *addInstance(InstanceOp instance, InstanceGraphNode *target);
139 
140 private:
141  friend class InstanceRecord;
142 
143  InstanceGraphNode() : module(nullptr) {}
144  InstanceGraphNode(const InstanceGraphNode &) = delete;
145 
146  /// Record that a module instantiates this module.
147  void recordUse(InstanceRecord *record);
148 
149  /// The module.
150  Operation *module;
151 
152  /// List of instance operations in this module. This member owns the
153  /// InstanceRecords, which may be pointed to by other InstanceGraohNode's use
154  /// lists.
155  InstanceList instances;
156 
157  /// List of instances which instantiate this module.
158  InstanceRecord *firstUse = nullptr;
159 
160  // Provide access to the constructor.
161  friend class InstanceGraph;
162 };

164 /// This graph tracks modules and where they are instantiated. This is intended
165 /// to be used as a cached analysis on FIRRTL circuits. This class can be used
166 /// to walk the modules efficiently in a bottom-up or top-down order.
167 ///
168 /// To use this class, retrieve a cached copy from the analysis manager:
169 /// auto &instanceGraph = getAnalysis<InstanceGraph>(getOperation());
170 class InstanceGraph {
171  /// This is the list of InstanceGraphNodes in the graph.
172  using NodeList = llvm::iplist<InstanceGraphNode>;
173 
174 public:
175  /// Create a new module graph of a circuit. This must be called on a FIRRTL
176  /// CircuitOp or MLIR ModuleOp.
177  explicit InstanceGraph(Operation *operation);
178 
179  /// Get the node corresponding to the top-level module of a circuit.
180  InstanceGraphNode *getTopLevelNode();
181 
182  /// Get the module corresponding to the top-lebel module of a circuit.
183  FModuleLike getTopLevelModule();
184 
185  /// Look up an InstanceGraphNode for a module. Operation must be an FModuleOp
186  /// or an FExtModuleOp.
187  InstanceGraphNode *lookup(Operation *op);
188 
189  /// Lookup an module by name.
190  InstanceGraphNode *lookup(StringAttr name);
191 
192  /// Lookup an InstanceGraphNode for a module. Operation must be an FModuleOp
193  /// or an FExtModuleOp.
194  InstanceGraphNode *operator[](Operation *op) { return lookup(op); }
195 
196  /// Look up the referenced module from an InstanceOp. This will use a
197  /// hashtable lookup to find the module, where
198  /// InstanceOp.getReferencedModule() will be a linear search through the IR.
199  Operation *getReferencedModule(InstanceOp op);
200 
201  /// Check if child is instantiated by a parent.
202  bool isAncestor(FModuleLike child, FModuleOp parent);
203 
204  /// Iterate through all modules.
205  using iterator = detail::AddressIterator<NodeList::iterator>;
206  iterator begin() { return nodes.begin(); }
207  iterator end() { return nodes.end(); }
208 
209  //===-------------------------------------------------------------------------
210  // Methods to keep an InstanceGraph up to date.
211  //
212  // These methods are not thread safe. Make sure that modifications are
213  // properly synchronized or performed in a serial context. When the
214  // InstanceGraph is used as an analysis, this is only safe when the pass is
215  // on a CircuitOp.
216 
217  /// Add a newly created module to the instance graph.
218  InstanceGraphNode *addModule(FModuleLike module);
219 
220  /// Remove this module from the instance graph. This will also remove all
221  /// InstanceRecords in this module. All instances of this module must have
222  /// been removed from the graph.
223  void erase(InstanceGraphNode *node);
224 
225  /// Replaces an instance of a module with another instance. The target module
226  /// of both InstanceOps must be the same.
227  void replaceInstance(InstanceOp inst, InstanceOp newInst);
228 
229  /// Returns pointer to member of operation list.
230  static NodeList InstanceGraph::*
231  getSublistAccessgetSublistAccess(Operation *) {
232  return &InstanceGraph::nodes;
233  }
234 
235 private:
236  InstanceGraph(const InstanceGraph &) = delete;
237 
238  InstanceGraphNode *topLevelNode;
239 
240  /// Get the node corresponding to the module. If the node has does not exist
241  /// yet, it will be created.
242  InstanceGraphNode *getOrAddNode(StringAttr name);
243 
244  /// The storage for graph nodes, with deterministic iteration.
245  NodeList nodes;
246 
247  /// This maps each operation to its graph node.
248  llvm::DenseMap<Attribute, InstanceGraphNode *> nodeMap;
249 };

254 template <typename T>
255 inline static T &formatInstancePath(T &into, const InstancePath &path) {
256  into << "$root";
257  for (auto inst : path)
258  into << "/" << inst.name() << ":" << inst.moduleName();
259  return into;
260 }
261 
262 template <typename T>
263 static T &operator<<(T &os, const InstancePath &path) {
264  return formatInstancePath(os, path);
