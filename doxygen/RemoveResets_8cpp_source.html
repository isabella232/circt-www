29 // Returns true if this value is invalidated. This requires that a value is
30 // only ever driven once. This is guaranteed if this runs after the
31 // `ExpandWhens` pass .
32 static bool isInvalid(Value val) {
33 
34  // Update `val` to the source of the connection driving `thisVal`. This walks
35  // backwards across users to find the first connection and updates `val` to
36  // the source. This assumes that only one connect is driving `thisVal`, i.e.,
37  // this pass runs after `ExpandWhens`.
38  auto updateVal = [&](Value thisVal) {
39  for (auto *user : thisVal.getUsers()) {
40  if (auto connect = dyn_cast<ConnectOp>(user)) {
41  if (connect.dest() != val)
42  continue;
43  val = connect.src();
44  return;
45  }
46  if (auto connect = dyn_cast<StrictConnectOp>(user)) {
47  if (connect.dest() != val)
48  continue;
49  val = connect.src();
50  return;
51  }
52  }
53  val = nullptr;
54  return;
55  };
56 
57  while (val) {
58  // The value is a port.
59  if (auto blockArg = val.dyn_cast<BlockArgument>()) {
60  FModuleOp op = cast<FModuleOp>(val.getParentBlock()->getParentOp());
61  auto direction = op.getPortDirection(blockArg.getArgNumber());
62  // Base case: this is an input port and cannot be invalidated in module
63  // scope.
64  if (direction == Direction::In)
65  return false;
66  updateVal(blockArg);
67  continue;
68  }
69 
70  auto *op = val.getDefiningOp();
71 
72  // The value is an instance port.
73  if (auto inst = dyn_cast<InstanceOp>(op)) {
74  auto resultNo = val.cast<OpResult>().getResultNumber();
75  // An output port of an instance crosses a module boundary. This is not
76  // invalid within module scope.
77  if (inst.getPortDirection(resultNo) == Direction::Out)
78  return false;
79  updateVal(val);
80  continue;
81  }
82 
83  // Base case: we found an invalid value. We're done, return true.
84  if (isa<InvalidValueOp>(op))
85  return true;
86 
87  // Base case: we hit something that is NOT a wire, e.g., a PrimOp. We're
88  // done, return false.
89  if (!isa<WireOp>(op))
90  return false;
91 
92  // Update `val` with the driver of the wire. If no driver found, `val` will
93  // be set to nullptr and we exit on the next while iteration.
94  updateVal(op->getResult(0));
95  };
96  return false;
97 }

99 void RemoveResetsPass::runOnOperation() {
100  LLVM_DEBUG(
101  llvm::dbgs() << "===----- Running RemoveResets "
102  "-----------------------------------------------===\n"
103  << "Module: '" << getOperation().getName() << "'\n";);
104 
105  bool madeModifications = false;
106  for (auto reg : llvm::make_early_inc_range(
107  getOperation().getBody()->getOps<RegResetOp>())) {
108 
109  // If the `RegResetOp` has an invalidated initialization, then replace it
110  // with a `RegOp`.
111  if (isInvalid(reg.resetValue())) {
112  LLVM_DEBUG(llvm::dbgs() << " - RegResetOp '" << reg.name()
113  << "' will be replaced with a RegOp\n");
114  ImplicitLocOpBuilder builder(reg.getLoc(), reg);
115  RegOp newReg =
116  builder.create<RegOp>(reg.getType(), reg.clockVal(), reg.name(),
117  reg.annotations(), reg.inner_symAttr());
118  reg.replaceAllUsesWith(newReg.getResult());
119  reg.erase();
120  madeModifications = true;
121  }
122  }
123 
124  if (!madeModifications)
125  return markAllAnalysesPreserved();
126 }
