53 InstanceGraph::InstanceGraph(Operation *operation) {
54  if (auto mod = dyn_cast<mlir::ModuleOp>(operation))
55  for (auto &op : *mod.getBody())
56  if ((operation = dyn_cast<CircuitOp>(&op)))
57  break;
58 
59  auto circuit = cast<CircuitOp>(operation);
60  auto topModuleName = circuit.nameAttr();
61  for (auto &op : *circuit.getBody()) {
62  auto module = dyn_cast<FModuleLike>(op);
63  if (!module)
64  continue;
65  auto name = module.moduleNameAttr();
66  auto *currentNode = getOrAddNode(name);
67  currentNode->module = module;
68  if (name == topModuleName)
69  topLevelNode = currentNode;
70  // Find all instance operations in the module body.
71  module.walk([&](InstanceOp instanceOp) {
72  // Add an edge to indicate that this module instantiates the target.
73  auto *targetNode = getOrAddNode(instanceOp.moduleNameAttr().getAttr());
74  currentNode->addInstance(instanceOp, targetNode);
75  });
76  }
77 }

118 void InstanceGraph::replaceInstance(InstanceOp inst, InstanceOp newInst) {
119  assert(inst.moduleName() == newInst.moduleName() &&
120  "Both instances must be targeting the same module");
121 
122  // Find the instance record of this instance.
123  auto *node = lookup(inst.moduleNameAttr().getAttr());
124  auto it = llvm::find_if(node->uses(), [&](InstanceRecord *record) {
125  return record->getInstance() == inst;
126  });
127  assert(it != node->usesEnd() && "Instance of module not recorded in graph");
128 
129  // We can just replace the instance op in the InstanceRecord without updating
130  // any instance lists.
131  (*it)->instance = newInst;
132 }

134 bool InstanceGraph::isAncestor(FModuleLike child, FModuleOp parent) {
135  DenseSet<InstanceGraphNode *> seen;
136  SmallVector<InstanceGraphNode *> worklist;
137  auto *cn = lookup(child);
138  worklist.push_back(cn);
139  seen.insert(cn);
140  while (!worklist.empty()) {
141  auto *node = worklist.back();
142  worklist.pop_back();
143  if (node->getModule() == parent)
144  return true;
145  for (auto *use : node->uses()) {
146  auto *mod = use->getParent();
147  if (!seen.count(mod)) {
148  seen.insert(mod);
149  worklist.push_back(mod);
150  }
151  }
152  }
153  return false;
154 }

156 ArrayRef<InstancePath> InstancePathCache::getAbsolutePaths(Operation *op) {
157  assert((isa<FModuleOp, FExtModuleOp>(op))); // extra parens makes parser smile
158 
159  // If we have reached the circuit root, we're done.
160  if (op == instanceGraph.getTopLevelNode()->getModule()) {
161  static InstancePath empty{};
162  return empty; // array with single empty path
163  }
164 
165  // Fast path: hit the cache.
166  auto cached = absolutePathsCache.find(op);
167  if (cached != absolutePathsCache.end())
168  return cached->second;
169 
170  // For each instance, collect the instance paths to its parent and append the
171  // instance itself to each.
172  SmallVector<InstancePath, 8> extendedPaths;
173  for (auto inst : instanceGraph[op]->uses()) {
174  auto instPaths = getAbsolutePaths(inst->getParent()->getModule());
175  extendedPaths.reserve(instPaths.size());
176  for (auto path : instPaths) {
177  extendedPaths.push_back(appendInstance(path, inst->getInstance()));
178  }
179  }
180 
181  // Move the list of paths into the bump allocator for later quick retrieval.
182  ArrayRef<InstancePath> pathList;
183  if (!extendedPaths.empty()) {
184  auto paths = allocator.Allocate<InstancePath>(extendedPaths.size());
185  std::copy(extendedPaths.begin(), extendedPaths.end(), paths);
186  pathList = ArrayRef<InstancePath>(paths, extendedPaths.size());
187  }
188 
189  absolutePathsCache.insert({op, pathList});
190  return pathList;
191 }
